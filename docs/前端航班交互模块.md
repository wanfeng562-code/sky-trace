## 2. 本地启动方法

**先启动后端**（需要另开终端，或确认后端已在运行）：
<!-- 注意修改路径 -->
```powershell
cd D:\Gitee\Sky-Trace\server
python -m uvicorn app.main:app --host 127.0.0.1 --port 8000
```

**再启动前端**：

```powershell
cd D:\Gitee\Sky-Trace\client
npm install          # 首次拉取依赖
npm run dev          # 启动 Vite 开发服务器，默认 http://localhost:5173
```

后端地址已在 `client/.env.development` 中配置好，无需修改：

```
VITE_API_BASE_URL=http://127.0.0.1:8000/api/v1
VITE_WS_URL=ws://127.0.0.1:8000/api/v1/ws/flights
```

后端接口文档（Swagger UI）：http://127.0.0.1:8000/docs

## 3. 后端接口一览

以下接口均已实现并验证，你直接调用即可。

### 3.1 航班列表（带筛选）

```
GET /api/v1/flights
```

| 参数        | 类型   | 说明                                               |
| ----------- | ------ | -------------------------------------------------- |
| `callsign`  | string | 按呼号模糊搜索，如 `CCA`                           |
| `bbox`      | string | 经纬度矩形，格式 `lon_min,lat_min,lon_max,lat_max` |
| `page`      | int    | 页码，默认 1                                       |
| `page_size` | int    | 每页条数，默认 100，最大 500                       |

**响应示例**：

```json
{
  "code": 0,
  "message": "ok",
  "data": {
    "total": 6821,
    "items": [
      {
        "flight_id": "abc1234567",
        "callsign": "CCA1234",
        "lat": 34.12,
        "lon": 108.93,
        "heading": 270.5,
        "speed_kts": 489.3,
        "altitude_ft": 35000,
        "updated_at": "2026-04-24T10:30:00Z"
      }
    ]
  }
}
```

**响应示例**：

```typescript
import type {

  FlightQueryParams,

  TrackPoint,

const api = axios.create({

export async function fetchFlights(params?: FlightQueryParams): Promise<FlightBrief[]> {
  const response = await api.get<ApiResponse<{ total: number; items: FlightBrief[] }>>(
    "/flights",

  );
  return response.data.data.items;
}

// 新增：拉取单航班详情
export async function fetchFlightDetail(flightId: string): Promise<FlightDetail> {
  const response = await api.get<ApiResponse<FlightDetail>>(

  return response.data.data;

export async function fetchFlightTrack(flightId: string): Promise<TrackPoint[]> {
```

```typescript
// ---- 新增：搜索与筛选状态 ----
const searchKeyword = ref<string>("")
const filterStatus = ref<"all" | "airborne" | "on_ground">("all")

// ---- 新增：航班详情 ----
const flightDetail = ref<FlightDetail | null>(null)
const detailLoading = ref(false)

// 修改 selectFlight：同时拉取详情
// 原来的 selectFlight 只拉轨迹，这里我们新增 loadFlightDetail
// 注意：不要修改原 selectFlight，新增独立方法
async function loadFlightDetail(flightId: string | null) {
  if (!flightId) {
    flightDetail.value = null
    return
  }
  detailLoading.value = true
  try {
    flightDetail.value = await fetchFlightDetail(flightId)
  } catch {
    flightDetail.value = null
  } finally {
    detailLoading.value = false
  }
}

// 前端筛选（在本地已有 flights 上过滤，无需每次请求后端）
const filteredFlights = computed(() => {
  let list = flights.value
  if (searchKeyword.value.trim()) {
    const kw = searchKeyword.value.trim().toUpperCase()
    list = list.filter(
      (f) =>
        f.flight_id.toUpperCase().includes(kw) ||
        (f.callsign ?? "").toUpperCase().includes(kw)
    )
  }
  if (filterStatus.value === "airborne") {
    list = list.filter((f) => (f.altitude_ft ?? 0) > 100)
  } else if (filterStatus.value === "on_ground") {
    list = list.filter((f) => (f.altitude_ft ?? 0) <= 100)
  }
  return list
})
```

## 8. Step 4 — 补充 WS 断线重连

**文件**：`src/services/ws.ts`

当前文件有 `// TODO: Add reconnect/backoff strategy` 注释。用以下内容**完整替换**该文件：

```typescript
import type { FlightBrief } from "../types/flight";

export type SnapshotHandler = (flights: FlightBrief[]) => void;
export type StatusHandler = (online: boolean) => void;

const RECONNECT_DELAYS = [2000, 4000, 8000, 15000, 30000]; // 退避序列（ms）

export function createFlightsSocket(
  onSnapshot: SnapshotHandler,
  onStatusChange?: StatusHandler
): { close: () => void } {
  let ws: WebSocket | null = null;
  let retryCount = 0;
  let closed = false; // 主动关闭标志，不再重连

  function connect() {
    ws = new WebSocket(import.meta.env.VITE_WS_URL);

    ws.onopen = () => {
      retryCount = 0;
      onStatusChange?.(true);
    };

    ws.onmessage = (event) => {
      try {
        const payload = JSON.parse(event.data) as {
          event: string;
          data: FlightBrief[];
        };
        if (payload.event === "snapshot") {
          onSnapshot(payload.data);
        }
      } catch {
        // 忽略格式错误的帧
      }
    };

    ws.onclose = () => {
      onStatusChange?.(false);
      if (!closed) {
        const delay = RECONNECT_DELAYS[Math.min(retryCount, RECONNECT_DELAYS.length - 1)];
        retryCount++;
        setTimeout(connect, delay);
      }
    };

    ws.onerror = () => {
      ws?.close(); // 触发 onclose 走重连
    };
  }

  connect();

  return {
    close() {
      closed = true;
      ws?.close();
    },
  };
}
```

> **注意**：这里改变了 `createFlightsSocket` 的返回值类型（从 `WebSocket` 变为 `{ close: () => void }`），需要同步修改 `stores/flight.ts` 中调用它的地方。

**`stores/flight.ts` 中对应修改**：

找到这段代码：


改为：

```typescript

```
